class Channel:

A fast, thread-safe message passing channel based on Rust's crossbeam-channel.

Supports bounded and unbounded capacities, blocking send/recv with timeouts, and Go-style multiplexed selection.

Examples:

from pysync import Channel ch = Channel(capacity=10) ch.send("hello") ch.recv() 'hello' ch.close()

Channel(capacity: int | None = None)

Initialize the Channel.

Args: capacity: Bounded capacity. If None, the channel is unbounded.

def send(self, item: Any, timeout: float | None = None) -> None:

Send an item to the channel. Blocks if bounded channel is full.

Args: item: The Python object to send. timeout: Optional maximum wait time in seconds.

Raises: ValueError: If the channel is closed. TimeoutError: If the send operation times out.

def recv(self, timeout: float | None = None) -> Any:

Receive an item from the channel. Blocks if the channel is empty.

Args: timeout: Optional maximum wait time in seconds.

Returns: The received Python object.

Raises: ValueError: If the channel is closed and empty. TimeoutError: If the receive operation times out.

Block and receive an item with a timeout. Releases the GIL.

def close(self) -> None:

Close the channel, preventing any further sends. Existing buffered items can still be received.

130        """
131        Remove and return a (key, value) pair from the dictionary.
132        Raises KeyError if the dictionary is empty.

class ThreadPool:

An OS-level physical thread pool scheduler for heavy parallel computing.

Avoids GIL contention and allows true parallel multi-threaded execution.

Warning: To prevent thread pool starvation deadlocks, do not submit nested tasks to the same ThreadPool and wait on their futures synchronously (e.g., via future.result()). Doing so under heavy concurrency will consume all worker threads, preventing nested tasks from ever running.

Examples:

from pysync import ThreadPool pool = ThreadPool(num_workers=4) future = pool.submit(lambda x: x * 2, 21) future.result() 42 pool.shutdown()

 4class ThreadGroup:
 5    """
 6    A structured concurrency thread manager.
 7    
 8    Ensures spawned thread lifetimes are securely bounded within the context manager block.
 9    Upon context exit, it blocks and joins all spawned child threads, including any threads
10    dynamically spawned by child tasks. Any exceptions raised in tasks are aggregated and
11    propagated as an ExceptionGroup.
12    
13    Examples:
14        >>> from pysync import ThreadGroup
15        >>> results = []
16        >>> with ThreadGroup() as tg:
17        ...     tg.spawn(lambda: results.append(1))
18        ...     tg.spawn(lambda: results.append(2))
19        >>> sorted(results)
20        [1, 2]
21    """
22    def __init__(self):
23        self._threads = []
24        self._errors = []
25        self._lock = threading.Lock()
26
27    def spawn(self, target, *args, **kwargs):
28        """
29        Spawn a new physical OS thread inside the thread group.
30        
31        Args:
32            target: The callable function to execute in the thread.
33            *args: Positional arguments to forward to the target.
34            **kwargs: Keyword arguments to forward to the target.
35            
36        Returns:
37            The spawned threading.Thread object.
38        """
39        def wrapper():
40            try:
41                target(*args, **kwargs)
42            except BaseException as e:
43                with self._lock:
44                    self._errors.append(e)
45
46        t = threading.Thread(target=wrapper)
47        t.start()
48        with self._lock:
49            self._threads.append(t)
50        return t
51
52    def __enter__(self):
53        return self
54
55    def __exit__(self, exc_type, exc_val, exc_tb):
56        joined = set()
57        while True:
58            with self._lock:
59                unjoined = [t for t in self._threads if t not in joined]
60                if not unjoined:
61                    break
62            for t in unjoined:
63                if t.is_alive() or t.ident is not None:
64                    t.join()
65                joined.add(t)
66
67        # If the main context block body raised an exception, prioritize it over thread exceptions.
68        if exc_val is not None:
69            return False
70
71        # Propagate the aggregated exceptions
72        if self._errors:
73            if len(self._errors) == 1:
74                raise self._errors[0]
75            raise ExceptionGroup("Multiple exceptions occurred inside ThreadGroup tasks", self._errors)

class Actor: 
 35class Actor:
 36    """
 37    An Actor model implementation guaranteeing thread-safe state isolation.
 38    
 39    All public methods and properties invoked on an Actor from external threads are intercepted
 40    and executed sequentially on a dedicated background OS thread via an internal Channel mailbox.
 41    Public state attributes cannot be directly read or mutated from outside the Actor thread,
 42    preventing data races in multi-threaded GIL-free environments.
 43    
 44    Supports lifecycle hooks `on_start()` and `on_stop()`, and supervision hook `on_error()`.
 45    
 46    Examples:
 47        >>> from pysync import Actor
 48        >>> class Counter(Actor):

110            # hook so users can observe/handle Actor startup failures, instead of
111            # swallowing the exception silently.
112            try:
113                self.on_error(e, "on_start", (), {})
114            except BaseException:
115                pass

118            try:

120                if msg is None:  # Shutdown sentinel
121                    break
122                method_name, args, kwargs, future = msg
123                try:

125                    method = object.__getattribute__(self, method_name)
126                    if callable(method):

128                    else:
129                        result = method
130                    if future is not None:

132                except BaseException as e:

136                    except BaseException:
137                        pass
